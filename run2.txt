#!/bin/bash
set -e

echo "=== VCF locations ==="
find /home/jupyter -name "AoU_srWGS_SV.v8.chr*.vcf.gz" -maxdepth 4 2>/dev/null | sort | head -25

echo ""
echo "=== All .sv_pgs_cache dirs ==="
find /home/jupyter -name ".sv_pgs_cache" -type d -maxdepth 5 2>/dev/null

echo ""
echo "=== Every file in every .sv_pgs_cache (filename + size only) ==="
for cache_dir in $(find /home/jupyter -name ".sv_pgs_cache" -type d -maxdepth 5 2>/dev/null); do
    echo "--- $cache_dir ---"
    find "$cache_dir" -type f -printf '%s %p\n' 2>/dev/null | sort -t/ -k6 | head -200
    echo "--- dirs ---"
    find "$cache_dir" -type d 2>/dev/null | head -50
done

echo ""
echo "=== tmp_parallel region file counts per dir ==="
for tmp_dir in $(find /home/jupyter -name "*.tmp_parallel" -type d -maxdepth 6 2>/dev/null); do
    n_geno=$(ls "$tmp_dir"/*.geno 2>/dev/null | wc -l)
    n_complete=$(ls "$tmp_dir"/region_*.geno 2>/dev/null | wc -l)
    echo "$tmp_dir: $n_geno geno files, $n_complete region geno files"
    ls "$tmp_dir"/ 2>/dev/null | head -10
    echo "..."
done

echo ""
echo "=== Python cache key check ==="
cd ~/SV-PGS && uv run python3 -c "
from pathlib import Path
from sv_pgs.io import _vcf_cache_key, _vcf_cache_dir, _find_any_complete_vcf_cache, _find_any_tmp_parallel
from sv_pgs.config import ModelConfig
config = ModelConfig()
vcf_path = list(Path('/home/jupyter').rglob('AoU_srWGS_SV.v8.chr1.vcf.gz'))[0]
print(f'VCF path: {vcf_path}')
print(f'Resolved: {vcf_path.resolve()}')
new_key = _vcf_cache_key(vcf_path, config)
print(f'Current cache key (no keep_indices): {new_key}')
cache_dir = _vcf_cache_dir(vcf_path)
print(f'Cache dir: {cache_dir}')
print(f'Cache dir exists: {cache_dir.exists()}')
if cache_dir.exists():
    legacy = _find_any_complete_vcf_cache(cache_dir)
    print(f'Legacy complete cache found: {legacy}')
    tmp = _find_any_tmp_parallel(cache_dir, new_key)
    print(f'Tmp parallel found: {tmp}')
    print(f'Tmp parallel exists: {tmp.exists()}')
    if tmp.exists():
        import os
        files = os.listdir(tmp)
        print(f'Tmp parallel contents ({len(files)} files): {files[:15]}')
    # List ALL files in cache dir
    all_files = sorted(cache_dir.iterdir())
    print(f'All cache dir entries ({len(all_files)}):')
    for f in all_files[:30]:
        sz = f.stat().st_size if f.is_file() else 0
        print(f'  {f.name}  ({"dir" if f.is_dir() else f"{sz:,} bytes"})')
"
